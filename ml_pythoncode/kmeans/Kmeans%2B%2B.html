def generate_points(npoints, radius):
    points = [Point() for _ in range(npoints)]

    # note: this is not a uniform 2-d distribution
    for p in points:
        r = random() * radius
        ang = random() * 2 * pi
        p.x = r * cos(ang)
        p.y = r * sin(ang)

    return points

def nearest_cluster_center(point, cluster_centers):
    """Distance and index of the closest cluster center"""
    def sqr_distance_2D(a, b):
        return (a.x - b.x) ** 2  +  (a.y - b.y) ** 2

    min_index = point.group
    min_dist = FLOAT_MAX

    for i, cc in enumerate(cluster_centers):
        d = sqr_distance_2D(cc, point)
        if min_dist > d:
            min_dist = d
            min_index = i

    return (min_index, min_dist)

def kpp(points, cluster_centers):
    cluster_centers[0] = copy(choice(points))
    d = [0.0 for _ in range(len(points))]

    for i in range(1, len(cluster_centers)):
        sum = 0
        for j, p in enumerate(points):
            d[j] = nearest_cluster_center(p, cluster_centers[:i])[1]
            # 将所有距离求和得到sum(D(x))
            sum += d[j]
        print("sum:  %s"%sum)
        sum *= random()
        print(" sum*random: %s"%sum)
        
        '''
        先取一个能落在Sum(D(x))中的随机值Randomm,sum-=D(x),直到 sum <=0,此时的点
        就是下一个种子点。
        '''
        for j, di in enumerate(d):
            sum -= di
            if sum > 0:
                continue
            cluster_centers[i] = copy(points[j])
            print("points[j].x:  ",cluster_centers[i].x)
            break

    for p in points:
        p.group = nearest_cluster_center(p, cluster_centers)[0]

def lloyd(points, nclusters):
    cluster_centers = [Point() for _ in range(nclusters)]
    print("before %s"%cluster_centers[0].x)
    # call k++ init
    kpp(points, cluster_centers)
    print("after %s" %cluster_centers[0].x)
    lenpts10 = len(points) >> 10

    '''
    初始化聚类中心，然后根据该聚类中心进行数据点分配
    '''
    changed = 0
    while True:
        # group element for centroids are used as counters
        for cc in cluster_centers:
            cc.x = 0
            cc.y = 0
            cc.group = 0

        for p in points:
            cluster_centers[p.group].group += 1
            cluster_centers[p.group].x += p.x
            cluster_centers[p.group].y += p.y

        for cc in cluster_centers:
            cc.x /= cc.group
            cc.y /= cc.group
        
        # 发现聚类中心的坐标，将所有点分配到固定的聚类中心。
        # find closest centroid of each PointPtr
        changed = 0
        for p in points:
            min_i = nearest_cluster_center(p, cluster_centers)[0]
            if min_i != p.group:
                changed += 1
                p.group = min_i

        # stop when 99.9% of points are good
        if changed <= lenpts10:
            break

    for i, cc in enumerate(cluster_centers):
        cc.group = i

    return cluster_centers

def kmeans(points, nclusters):
    cluster_centers = [copy(choice(points)) for _ in range(nclusters)]
    lenpts10 = len(points) >> 10
    changed = 0
    while True:
        for p in points:
            # 聚类中心 数目加1， 聚类中心x叠加，聚类中心y叠加
            cluster_centers[p.group].group += 1
            cluster_centers[p.group].x += p.x
            cluster_centers[p.group].y += p.y

        for cc in cluster_centers:
            #print cc.x,cc.y,cc.group
            # 聚类中心平均值
            cc.x /= cc.group
            cc.y /= cc.group
        changed = 0
        for p in points:
            # 返回p.group  也就是聚类中心
            min_i = nearest_cluster_center(p, cluster_centers)[0]
            if min_i != p.group:
                changed += 1
                p.group = min_i

        # stop when 99.9% of points are good
        if changed <= lenpts10:
            break
    for i, cc in enumerate(cluster_centers):
        cc.group = i

    return cluster_centers
